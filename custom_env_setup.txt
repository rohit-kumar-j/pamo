# 1. Create conda env and activate
conda create -n pamo python=3.10 -y
conda activate pamo

# 2. Install CUDA toolkit via conda
conda install -c nvidia cuda-toolkit=12.8 -y

# 3. Set CUDA_HOME and make it persistent across sessions
export CUDA_HOME=$CONDA_PREFIX
mkdir -p $CONDA_PREFIX/etc/conda/activate.d
echo 'export CUDA_HOME=$CONDA_PREFIX' > $CONDA_PREFIX/etc/conda/activate.d/cuda_env.sh

# 4. Create /usr/local/cuda symlinks (conda puts headers/libs in non-standard paths)
sudo rm -rf /usr/local/cuda
sudo mkdir -p /usr/local/cuda
sudo ln -s $CONDA_PREFIX/targets/x86_64-linux/include /usr/local/cuda/include
sudo ln -s $CONDA_PREFIX/targets/x86_64-linux/lib /usr/local/cuda/lib64
sudo ln -s $CONDA_PREFIX/bin /usr/local/cuda/bin

# 5. Install PyTorch with matching CUDA 12.8
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu128

# 6. Install setuptools (needed for pkg_resources)
pip install setuptools

# 7. Install pdmc separately with --no-build-isolation (avoids pkg_resources error)
pip install pdmc --no-build-isolation

# 8. Clone pamo and patch the warp build script (static lib → shared lib)
git clone https://github.com/SarahWeiii/pamo.git
cd pamo
sed -i 's/nvrtc_static/nvrtc/g' simp_cuda/safe_project/warp_/warp/build_dll.py
sed -i 's/nvrtc-builtins_static/nvrtc-builtins/g' simp_cuda/safe_project/warp_/warp/build_dll.py

# 9. Run setup
bash setup.sh

# 10. Verify
python -c "import pamo; import torchcumesh2sdf; import pdmc; print('All imports OK')"
